fun firstkprimes(){
    var *primes = 2000;
    var i = 2;
    while(i <= 1000){
        if(*(primes + i) == 0){
            var j = i;
            while(j <= 1000){
                *(primes + j) = 1;
                j = j + i;
            }
            asm{
                out (i)
            };
        }
        i = i + 1;
    }
}
