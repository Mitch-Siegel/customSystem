fun fib(var n){
    var *fibarr = 256;
    *fibarr = 0;
    *(fibarr + 1) = 1;
    var j;
    var i = 2;

    while(i <= n){
        j = *(fibarr + i - 1) + *(fibarr + i - 2);
        *(fibarr + i) = j;
        i = i + 1;
    }
    i = 1;
    while(i <= n){
        print(*(fibarr + i));
        i = i + 1;
    }
    
}
