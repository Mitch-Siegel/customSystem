
# this function computes the nth fibonacci number 
fun fib(var n)
{
    if(n < 2)
        return n;
    else
        return fib(n - 1) + fib(n - 2);
    
}

# find all fibonacci numbers up to and including n
fun firstnfibs(var n){
    var i = 1;
    var j;
    var result;
    while(i <= n){
        j = n - (i - 1);
        result = fib(i);
        result = fib(j);

        i = i + 1;
    }

}
