asm
{
    #include "CPU.asm"
    entry code
    code:
    push $20
    call firstnfibs
    hlt
};


// compute the first n fibonacci numbers (using pointers!)
fun firstnfibs(var n){
    var *fibarr = 1000;
    *fibarr = 0;
    *(fibarr + 1) = 1;
    var j;

    var i = 2;
    while(i <= n){
        j = *(fibarr + i - 1) + *(fibarr + i - 2);
        asm{
            out [j]
        };
        *(fibarr + i) = j;
        i = i + 1;
    }
}
